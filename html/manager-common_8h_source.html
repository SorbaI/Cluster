91static bool manager_init_socket(INFO_MANAGER* manager)
92{
93 if (manager->is_init == false) {

95 return false;
96 }
97 // Создаём сокет, слушающий подключения клиентов.

99 if (manager->listen_sock_fd == -1)

101 fprintf(stderr, "[manager_init] Unable to create socket!\n");

103 }
104

110 return false;
111 }

117 }
118
119 // Активируем очередь запросов на подключение.
120 if (listen(manager->listen_sock_fd, manager->num_nodes /* Размер очереди запросов на подключение */) == -1)

140 DEBUG("Wait for worker_node to connect\n");
